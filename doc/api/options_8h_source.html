15 #include <boost/program_options.hpp>
16 #include <string>
17 #include <iostream>

20 namespace bpo = boost::program_options;
21 
23 auto handle_options(int argc, char* argv[])
24 {
25  // Declare a group of options that will be allowed only on command line
26  bpo::options_description generic_options{"Command-line-only options"};
27  generic_options.add_options()
28  ("help,h", "help screen")
29  ("verbose,v", "verbose mode")
30  ("version", "software version")
31  ("config", bpo::value<std::string>()->required(), "configuration file")
32  ;
33 
34  // Declare a simpler way to call on command line
35  bpo::positional_options_description positional_options;
36  positional_options.add("config", 1);
37 
38  // Allowed both on command line and in config file
39  bpo::options_description model_options{"Propagation model parameters"};
40  model_options.add_options()
41  ("lon_0", bpo::value<double>(), "Origin point longitude")
42  ("lat_0", bpo::value<double>(), "Origin point latitude")
43  ("alt_0", bpo::value<double>(), "Number of gene copies at introduction point")
44  ("duration", bpo::value<int>(), "Number of generations to simulate")
45  ;
46 
47  bpo::options_description command_line_options;
48  command_line_options.add(generic_options).add(model_options);
49 
50  bpo::options_description file_options{"General options (command line values will overwrite congif file values)"};
51  file_options.add(model_options);
52 
53  bpo::variables_map vm;
54 
55  try
56  {
57 
58  bpo::store(
59  bpo::command_line_parser(argc, argv).
60  options(command_line_options).
61  positional(positional_options).
62  run(), vm); // can throw
63 
64  // --help option
65  if (vm.count("help"))
66  {
67  std::cout << "--------------------------------------------------------------------------------------" << std::endl;
68  std::cout << "| This is SpOCK propagator. |" << std::endl;
69  std::cout << "| - Purpose: propagates satellites orbits. |" << std::endl;
70  std::cout << "| - Author: Aaron Ridley, Aidan Kingwell, Arnaud Becheler 2022. |" << std::endl;
71  std::cout << "| - Usage: " << argv[0] << " [options] <config> ... |" << std::endl;
72  std::cout << "--------------------------------------------------------------------------------------|" << std::endl;
73  std::cout << "\n" << generic_options << std::endl;
74  std::cout << "\n" << file_options << std::endl;
75  return vm;
76  // SUCCESS
77  }
78  // --version option
79  if (vm.count("version"))
80  {
81  std::cout << "SpOCK version 0.1" << std::endl;
82  return vm;
83  // SUCCESS
84  }
85  bpo::notify(vm); // throws on error, so do after help in case there are any problems
86  }
87  catch(boost::program_options::required_option& e)
88  {
89  throw;
90  }
91  catch(boost::program_options::error& e)
92  {
93  throw;
94  }
95  if (vm.count("config"))
96  {
97  std::ifstream ifs{vm["config"].as<std::string>().c_str()};
98  if (ifs){
99  store(parse_config_file(ifs, file_options), vm);
100  }
101  }
102  notify(vm);
103  return vm;
104 } // end of handle_options

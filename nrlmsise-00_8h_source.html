33 struct nrlmsise_flags {
34  int switches[24];
35  double sw[24];
36  double swc[24];
37 };
38 /*
39  * Switches: to turn on and off particular variations use these switches.
40  * 0 is off, 1 is on, and 2 is main effects off but cross terms on.
41  *
42  * Standard values are 0 for switch 0 and 1 for switches 1 to 23. The
43  * array "switches" needs to be set accordingly by the calling program.
44  * The arrays sw and swc are set internally.
45  * 3 to 8, and 14 try 0
46  * switches[i]:
47  * i - explanation
48  * -----------------
49  * 0 - output in meters and kilograms instead of centimeters and grams
50  * 1 - F10.7 effect on mean
51  * 2 - time independent
52  * 3 - symmetrical annual
53  * 4 - symmetrical semiannual
54  * 5 - asymmetrical annual
55  * 6 - asymmetrical semiannual
56  * 7 - diurnal
57  * 8 - semidiurnal
58  * 9 - daily ap [when this is set to -1 (!) the pointer
59  * ap_a in struct nrlmsise_input must
60  * point to a struct ap_array]
61  * 10 - all UT/long effects
62  * 11 - longitudinal
63  * 12 - UT and mixed UT/long
64  * 13 - mixed AP/UT/LONG
65  * 14 - terdiurnal
66  * 15 - departures from diffusive equilibrium
67  * 16 - all TINF var
68  * 17 - all TLB var
69  * 18 - all TN1 var
70  * 19 - all S var
71  * 20 - all TN2 var
72  * 21 - all NLB var
73  * 22 - all TN3 var
74  * 23 - turbo scale height var
75  */

93 struct nrlmsise_input {
94  int year; /* year, currently ignored */
95  int doy; /* day of year */
96  double sec; /* seconds in day (UT) */
97  double alt; /* altitude in kilometers */
98  double g_lat; /* geodetic latitude */
99  double g_long; /* geodetic longitude */
100  double lst; /* local apparent solar time (hours), see note below */
101  double f107A; /* 81 day average of F10.7 flux (centered on doy) */
102  double f107; /* daily F10.7 flux for previous day */
103  double ap; /* magnetic index(daily) */
104  struct ap_array *ap_a; /* see above */
105 };
106 /*
107  * NOTES ON INPUT VARIABLES:
108  * UT, Local Time, and Longitude are used independently in the
109  * model and are not of equal importance for every situation.
110  * For the most physically realistic calculation these three
111  * variables should be consistent (lst=sec/3600 + g_long/15).
112  * The Equation of Time departures from the above formula
113  * for apparent local time can be included if available but
114  * are of minor importance.
115  *
116  * f107 and f107A values used to generate the model correspond
117  * to the 10.7 cm radio flux at the actual distance of the Earth
118  * from the Sun rather than the radio flux at 1 AU. The following
119  * site provides both classes of values:
120  * ftp://ftp.ngdc.noaa.gov/STP/SOLAR_DATA/SOLAR_RADIO/FLUX/
121  *
122  * f107, f107A, and ap effects are neither large nor well
123  * established below 80 km and these parameters should be set to
124  * 150., 150., and 4. respectively.
125  */

133 struct nrlmsise_output {
134  double d[9]; /* densities */
135  double t[2]; /* temperatures */
136 };
137 /*
138  * OUTPUT VARIABLES:
139  * d[0] - HE NUMBER DENSITY(CM-3)
140  * d[1] - O NUMBER DENSITY(CM-3)
141  * d[2] - N2 NUMBER DENSITY(CM-3)
142  * d[3] - O2 NUMBER DENSITY(CM-3)
143  * d[4] - AR NUMBER DENSITY(CM-3)
144  * d[5] - TOTAL MASS DENSITY(GM/CM3) [includes d[8] in td7d]
145  * d[6] - H NUMBER DENSITY(CM-3)
146  * d[7] - N NUMBER DENSITY(CM-3)
147  * d[8] - Anomalous oxygen NUMBER DENSITY(CM-3)
148  * t[0] - EXOSPHERIC TEMPERATURE
149  * t[1] - TEMPERATURE AT ALT
150  *
151  *
152  * O, H, and N are set to zero below 72.5 km
153  *
154  * t[0], Exospheric temperature, is set to global average for
155  * altitudes below 120 km. The 120 km gradient is left at global
156  * average value for altitudes below 72 km.
157  *
158  * d[5], TOTAL MASS DENSITY, is NOT the same for subroutines GTD7
159  * and GTD7D
160  *
161  * SUBROUTINE GTD7 -- d[5] is the sum of the mass densities of the
162  * species labeled by indices 0-4 and 6-7 in output variable d.
163  * This includes He, O, N2, O2, Ar, H, and N but does NOT include
164  * anomalous oxygen (species index 8).
165  *
166  * SUBROUTINE GTD7D -- d[5] is the "effective total mass density
167  * for drag" and is the sum of the mass densities of all species
168  * in this model, INCLUDING anomalous oxygen.
169  */
